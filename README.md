## Usage

```shell
npm install @flodejs/json-schema-editor-arco
```

```jsx
import { useState } from 'react';
import JsonSchemaEditor from '@flodejs/json-schema-editor-arco';

export default () => {
  const [jsonData, setJsonData] = useState({});
  return (
    <JsonSchemaEditor
      data={jsonData}
      onChange={(data) => {
        setJsonData(data);
      }}
    />
  );
};
```
